#### Dynamic Sitemap Generator (JavaScript)
Create `sitemap-generator.js`:
```javascript
// Generate dynamic sitemap from Supabase products
async function generateDynamicSitemap() {
  const { data: products, error } = await supabase
    .from('products')
    .select('id, name, updated_at')
    .eq('status', 'active');
  
  if (error) {
    console.error('Error fetching products:', error);
    return;
  }
  
  let sitemap = '<?xml version="1.0" encoding="UTF-8"?>\n';
  sitemap += '<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">\n';
  
  // Add static pages
  sitemap += '  <url>\n';
  sitemap += '    <loc>https://shopup.gh/</loc>\n';
  sitemap += '    <changefreq>daily</changefreq>\n';
  sitemap += '    <priority>1.0</priority>\n';
  sitemap += '  </url>\n';
  
  // Add product pages
  products.forEach(product => {
    const slug = product.name.toLowerCase().replace(/\s+/g, '-');
    sitemap += '  <url>\n';
    sitemap += `    <loc>https://shopup.gh/products/${slug}</loc>\n`;
    sitemap += `    <lastmod>${new Date(product.updated_at).toISOString().split('T')[0]}</lastmod>\n`;
    sitemap += '    <changefreq>weekly</changefreq>\n';
    sitemap += '    <priority>0.8</priority>\n';
    sitemap += '  </url>\n';
  });
  
  sitemap += '</urlset>';
  
  return sitemap;
}

// Download sitemap
async function downloadSitemap() {
  const sitemap = await generateDynamicSitemap();
  const blob = new Blob([sitemap], { type: 'text/xml' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = 'sitemap.xml';
  a.click();
}
```

```javascript
async function optimizeProductImages() {
  const { data: products } = await supabase
    .from('products')

  products.forEach(product => {
```

### 3. Canonical URLs & Duplicate Content Prevention

#### Implementation Guide
```javascript
// Prevent duplicate content issues
function setCanonicalURL() {
  // Get current URL
  const currentURL = window.location.href;
  
  // Remove query parameters for canonical
  const canonicalURL = currentURL.split('?')[0].split('#')[0];
  
  // Set canonical
  let canonical = document.querySelector('link[rel="canonical"]');
  if (!canonical) {
    canonical = document.createElement('link');
    canonical.rel = 'canonical';
    document.head.appendChild(canonical);
  }
  canonical.href = canonicalURL;
}

// Handle URL parameters (filters, sorting)
function handleURLParameters() {
  const url = new URL(window.location.href);
  const params = url.searchParams;
  
  // If has filter/sort parameters, add noindex
  if (params.has('sort') || params.has('filter')) {
    let meta = document.createElement('meta');
    meta.name = 'robots';
    meta.content = 'noindex, follow';
    document.head.appendChild(meta);
  }
}

// Call on page load
document.addEventListener('DOMContentLoaded', () => {
  setCanonicalURL();
  handleURLParameters();
});
```

```javascript
// Add to all pages after GA4 initialization

// 1. Product View
function trackProductView(product) {
  gtag('event', 'view_item', {
    currency: 'GHS',
    value: product.price,
    items: [{
      item_id: product.id,
      item_name: product.name,
      item_category: product.category,
      price: product.price
    }]
  });
}

// 2. Add to Cart
function trackAddToCart(product, quantity) {
  gtag('event', 'add_to_cart', {
    currency: 'GHS',
    value: product.price * quantity,
    items: [{
      item_id: product.id,
      item_name: product.name,
      item_category: product.category,
      quantity: quantity,
      price: product.price
    }]
  });
}

// 3. Begin Checkout
function trackBeginCheckout(cartItems, totalValue) {
  gtag('event', 'begin_checkout', {
    currency: 'GHS',
    value: totalValue,
    items: cartItems
  });
}

// 4. Purchase
function trackPurchase(order) {
  gtag('event', 'purchase', {
    transaction_id: order.id,
    value: order.total,
    currency: 'GHS',
    tax: order.tax,
    shipping: order.shipping_cost,
    items: order.items
  });
}

// 5. Search
function trackSearch(searchTerm, results) {
  gtag('event', 'search', {
    search_term: searchTerm,
    results_count: results.length
  });
}

// 6. Custom: Seller Registration
function trackSellerSignup() {
  gtag('event', 'seller_signup', {
    event_category: 'engagement',
    event_label: 'new_seller'
  });
}
```
